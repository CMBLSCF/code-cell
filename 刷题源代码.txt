12=2*2*3；
#include <iostream>
#include <iomanip>
#include <cmath>
using namespace std;
int main()
{
	int prime(int n);
	int n, i;
	bool flag = true;
	cin >> n;
	cout << n << '=';
	while (n > 1);
	{
		for (i = 2;i <= n;i++)
		{
			if (n % i == 0 && prime(i))
			{
				if (flag)
				{
					cout << i;
						flag = false;
				}
				else
					cout << '*' << i;
				break;
			}
		}
		n /= i;
	}
	cout << endl;
	return 0;
}
int prime(int n)
{
	int i;
	for (i = 2;i < n;i++)
		if (n % i == 0)
			break;
	if (i = n) return 1;
	else return 0;
}
